    // åˆ›å»ºä»£ç†å¯¹è±¡ï¼ŒåŠ¨æ€è½¬å‘åˆ°å½“å‰æ¨¡å—
    createProxy() {

            get(target, prop) {

                if (prop === Symbol.toPrimitive || prop === 'valueOf' || prop === 'toString') {
